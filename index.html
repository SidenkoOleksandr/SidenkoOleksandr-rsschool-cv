            function insertionSort($unsortedArr) {
              $array = $unsortedArr;
              $n = count($array);
              for ($i = 1; $i < $n; $i++) {
                  $val = $array[$i];
                  $k = $i;
                  while (($k > 0) && ($array[$k - 1] > $val)) {
                      $array[$k] = $array[$k - 1];
                      $k = $k - 1;
                  }
                  $array[$k] = $val;
              }
              return $array;
            }
